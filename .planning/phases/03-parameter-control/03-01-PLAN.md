/// Get the human-readable display string for a parameter value.
pub fn get_parameter_display(&amp;self, id: u32) -&gt; Result&lt;String, HostError&gt; {
    let ctrl = self.controller.as_ref()
        .ok_or_else(|| HostError::InvalidState("no edit controller".to_string()))?;

    let normalized_value = unsafe { ctrl.getParamNormalized(id) };

    // Get human-readable string from plugin
    let mut string128: [u16; 128] = [0; 128];
    let result = unsafe {
        ctrl.getParamStringByValue(id, normalized_value, string128.as_mut_ptr())
    };

    if result == kResultOk {
        Ok(string128_to_string(&amp;string128))
    } else {
        // Fallback: return normalized value as string
        Ok(format!("{:.3}", normalized_value))
    }
}
```

impl ParamInfo {
    /// Check if this parameter can be written by the host.
    ///
    /// A writable parameter must have kCanAutomate flag AND NOT have kIsReadOnly flag.
    pub fn is_writable(&amp;self) -&gt; bool {
        const K_CAN_AUTOMATE: u32 = 1 &lt;&lt; 0;
        const K_IS_READ_ONLY: u32 = 1 &lt;&lt; 1;

        (self.flags &amp; K_CAN_AUTOMATE != 0) &amp;&amp; (self.flags &amp; K_IS_READ_ONLY == 0)
    }

    /// Check if this parameter should be hidden from UI.
    pub fn is_hidden(&amp;self) -&gt; bool {
        const K_IS_HIDDEN: u32 = 1 &lt;&lt; 5;
        self.flags &amp; K_IS_HIDDEN != 0
    }

    /// Check if this is a bypass parameter.
    pub fn is_bypass(&amp;self) -&gt; bool {
        const K_IS_BYPASS: u32 = 1 &lt;&lt; 4;
        self.flags &amp; K_IS_BYPASS != 0
    }

    /// Check if this parameter is read-only.
    pub fn is_read_only(&amp;self) -&gt; bool {
        const K_IS_READ_ONLY: u32 = 1 &lt;&lt; 1;
        self.flags &amp; K_IS_READ_ONLY != 0
    }
}
```

Flag constants from VST3 spec:
- kCanAutomate = 1 &lt;&lt; 0 (parameter can be automated)
- kIsReadOnly = 1 &lt;&lt; 1 (parameter cannot be changed by host)
- kIsBypass = 1 &lt;&lt; 4 (bypass on/off parameter)
- kIsHidden = 1 &lt;&lt; 5 (parameter should not be shown in UI)

These helpers enable filtering parameters before exposing to AI in Phase 4.
